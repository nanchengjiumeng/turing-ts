命令名称 Media_GetVolume 获取声音大小
命令功能 获取声音大小
语法格式 Volume = TURING.Media_GetVolume()
参数说明 无
返回值 Volume:整数型，音量（最大值：100）
按键精灵
复制代码 
'在线音乐播放
Call TURING.Media_PlayMusic("http://www.turing.video/img/music.mp3")
'判断当前是否播放
TracePrint TURING.Media_IsPlaying()
'获取声音大小
声音大小 = TURING.Media_GetVolume()
TracePrint 声音大小
'设置声音大小
Call TURING.Media_SetVolume(声音大小/2)
TracePrint TURING.Media_GetVolume()
'获取媒体的长度
播放时间 = TURING.Media_GetMusicLength()
TracePrint 播放时间
i = 0
Do 
    Delay 1000
    '获取当前播放进度
    播放进度 = TURING.Media_GetMusicPos()
    TracePrint 播放进度
    If 播放时间 = 播放进度 Then 
        Exit Do
    End If
    i = i + 1
    If i = 10 Then   '等10秒后暂停
        '设置当前播放进度
        Call TURING.Media_SetMusicPos(播放时间 / 2)
        '暂停播放
        Call TURING.Media_PauseMusic()
    ElseIf i = 13 Then '等3秒后再继续
        Call TURING.Media_SetVolume(声音大小)
        '继续播放
        Call TURING.Media_ResumeMusic()
    End If
Loop
'判断当前是否播放
TracePrint TURING.Media_IsPlaying()

EndScript
'脚本结束后，记得要停止音乐播放，不然声音会卡进程!
Sub OnScriptExit()
    Call TURING.Media_StopMusic() 
End Sub
